{{/* * This shortcode extends figure shortcode from the Hugo. * It adds the ability to click on the image to view it in fullscreen. * It adds support for theme images (light/dark). * If image does not have a theme variant, only provide src attribute. * If image has a theme variant, provide data-src-light and data-src-dark attributes and set the src attribute to the default variant. */}} {{- if .Get "link" -}} {{- end -}} {{- $light_u := urls.Parse (.Get "data-src-light") -}} {{- $light_src := $light_u.String -}} {{- if not $light_u.IsAbs -}} {{- with or (.Page.Resources.Get $light_u.Path) (resources.Get $light_u.Path) -}} {{- $light_src = .RelPermalink -}} {{- end -}} {{- end -}} {{- $dark_u := urls.Parse (.Get "data-src-dark") -}} {{- $dark_src := $dark_u.String -}} {{- if not $dark_u.IsAbs -}} {{- with or (.Page.Resources.Get $dark_u.Path) (resources.Get $dark_u.Path) -}} {{- $dark_src = .RelPermalink -}} {{- end -}} {{- end -}} {{- $u := urls.Parse (.Get "src" | default "") -}} {{- $src := $u.String -}} {{- if not $u.IsAbs -}} {{- with or (.Page.Resources.Get $u.Path) (resources.Get $u.Path) -}} {{- $src = .RelPermalink -}} {{- end -}} {{- end -}} {{- if .Get "link" }}{{ end -}} {{- if or (or (.Get "title") (.Get "caption")) (.Get "attr") -}} {{- end }}
